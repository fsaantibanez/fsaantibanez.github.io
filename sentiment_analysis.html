def dt_hiperparams(model,max_leaf_nodes,min_samples_split):
    heatmap_X = model.cv_results_[max_leaf_nodes].data.astype(np.float)
    heatmap_Y = model.cv_results_[min_samples_split].data.astype(np.float)
    heatmap_Z = model.cv_results_['mean_test_score']
    heatmap_df = pd.DataFrame(dict(max_leaf_nodes = heatmap_X, min_samples_split = heatmap_Y, test_AUC = heatmap_Z))
    heatmap_df_pivot = heatmap_df.pivot_table(index = 'max_leaf_nodes', columns = 'min_samples_split', values ='test_AUC')
    f, ax = plt.subplots(figsize=(7, 3))
    sns.heatmap(heatmap_df_pivot, annot=True, fmt="0.4f", linewidths=.5,
            ax=ax, cmap='GnBu', robust=True, xticklabels=heatmap_df_pivot.columns.values.round(2),
            yticklabels=heatmap_df_pivot.index.values.round(2))
    plt.show();

def hiperparams_matrix(modelo,param_learning_rate,param_n_estimators):
    heatmap_X = modelo.cv_results_[param_learning_rate].data.astype(np.float)
    heatmap_Y = modelo.cv_results_[param_n_estimators].data.astype(np.float)
    heatmap_Z = modelo.cv_results_['mean_test_score']
    heatmap_df = pd.DataFrame(dict(learning_rate = heatmap_X, n_estimators = heatmap_Y, test_AUC = heatmap_Z))
    heatmap_df_pivot = heatmap_df.pivot_table(index = 'learning_rate', columns = 'n_estimators', values ='test_AUC')
    f, ax = plt.subplots(figsize=(7, 3))
    sns.heatmap(heatmap_df_pivot, annot=True, fmt="0.4f", linewidths=.5,
                ax=ax, cmap='GnBu', robust=True, xticklabels=heatmap_df_pivot.columns.values.round(2),
                yticklabels=heatmap_df_pivot.index.values.round(2))
    plt.show();
